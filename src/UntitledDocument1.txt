export const ThreeButton = () => {
    const classes = useStyles();
    useEffect(() => {
        aos.init({
            easing: 'ease-out-back',
            duration: 1200,
        });
    }, [])
    const [state, setState] = useState(portfolio)
    const backgroundColor = {
        border: '1px solid black',
        backgroundColor: '#fff',
        color: '#000'
    }
    const backgroundColor1 = {
        border: 'none',
        backgroundColor: '#168ACB',
        color: '#fff'
    }
    const filterItem = (result) => {
        const response = portfolio.filter((item) => {
            return item.stage === result
        })
        setState(response)
    }
    return (
        <Box className={classes.main}>
            <Box className={classes.buttonBox}>
                <button style={state.length == 19 ? backgroundColor1 : backgroundColor} className={classes.button} data-aos="fade-up" onClick={() => setState(portfolio)}>All</button>
                <button style={state.length == 10 ? backgroundColor1 : backgroundColor} className={classes.button} data-aos="fade-up" onClick={() => filterItem('Early')}>Early</button>
                <button style={state.length == 9 ? backgroundColor1 : backgroundColor} className={classes.button} data-aos="fade-up" onClick={() => filterItem('Growth')}>Growth</button>
            </Box>
            <Grid container xs={12} spacing={2} className={classes.grid}>
                {
                    state.map((item, index) => {
                        const { logo, companyName, Sector, stage, KeyInvestors } = item
                        return (
                            <Grid item xs={12} sm={6} md={4} data-aos="fade-up" key={index}>
                                <Box className={classes.allBox}>
                                    <Box style={{ display: 'flex', justifyContent: 'space-between' }}>
                                        <Box>
                                            <img className={classes.image} src={logo} alt={companyName} />
                                            <h1 className={classes.companyName}>{companyName}</h1>
                                            <h1 className={classes.Sector}>{Sector}</h1>
                                        </Box>
                                        <Box>
                                            <h1 className={classes.set}>{stage}</h1>
                                        </Box>
                                    </Box>
